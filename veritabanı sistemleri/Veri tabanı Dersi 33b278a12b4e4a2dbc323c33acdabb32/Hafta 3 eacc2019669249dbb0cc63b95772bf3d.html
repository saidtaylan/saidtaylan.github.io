// Biz ise veritabanında toplam kaç ülkeden kişi olduğunu görmek istiyoruz.

Sonuç kümesi: Sorgu sonucunda dönen sonuç tablosudur.
"<>", "!=" -> eşit değildir anlamına gelir.
select * from yazarlar where isim <> 'Ali'
// ismi 'Ali' olmayanları getirir

select * from yazarlar where soyisim < 'Kaya'
// soyismi alfabetik olarak 'Kaya'dan küçük olanları getirir.

select * from yazarlar where dogum_yili <> 1910
// doğum yılı 1910 olmayan yazarları getir.

select * from yazarlar where isim > 'Ali'
// alfabetik olarak isminin harfleri sırasıyla 'Ali'den büyük olanları getirir.

select * from yazarlar where dogum_yili>1825 and dogum_yili<1943

select * from yazarlar where dogum_yili between 1850 and 1900
// dogum yılı 1850 ile 1900 yılları arasındaki yazarları getirir.
// Between; girilen sütun değeri, girilen sayısal aralığa eşit olan satırları getirir
                             --   AND OR  --

select * from uyeler where isim = 'Deniz' and cinsiyet = 'K'
// ismi Deniz VE cinsiyeti Kadın olan üyeleri getirir
// AND ifadesi ile birden fazla şart koyabildik. Sadece ismi Deniz VE cinsiyeti KADIN olanları döndürür

select * from uyeler where isim = 'Deniz' or cinsiyet = 'K'
// ismi Deniz veya cinsiyeti Kadın olanları getirir
// Üyenin ya isminin Deniz olması ya cinsiyetinin Kadın olması ya da hem Deniz hem de Kadın olması gerekir.
// Yani OR ifadesinde şartlarıdan birinin doğru olması sonuçların getirilmesi için yeterlidir.
// Ancak arada AND bağlacı olmasaydı ve ismi Deniz olan ama cinsiyeti Kadın olan hiçbir üye olmasaydı hiçbir veri gelmeyecekti 

select * from uyeler where cinsiyet='E' and (soyisim = 'Aydın' or soyisim='Zafer')
// cinsiyeti kesinlikle Erkek olan, soyismi 'Aydın' veya 'Zafer' olanları getirir. 

select * from uyeler where (yas>30 and cinsiyet='E') or (yas<25 and cinsiyet='K')
// yaşı 30'dan büyük ve erkek VEYA yaşı 25'den küçük olan ve kadın olanları getirir

select * from personel where isim = 'Ahmet' or isim = 'Mehmet'
// ismi Mehmet VEYA Ahmet olanları getirir.
// iki şartın birden sağlanmasına ihtiyaç yoktur.

select * from where(yas>30 or sehir='Ankara') and (maas>2500 or cinsiyet='K')

                              --  ORDER BY  --

// tanım: Verileri, kendisinden sonra yazılan sütuna göre sıralayıp getirir.

select * from personel order by isim asc
// uyeleri seçip uyeleri isimlerine göre alfabetik şekilde sıralar
// asc: ascending yani 'artan sıra'/'küçükten büyüğe' demektir.
// Select komutu zaten sıralı bir şekilde verileri getirdiği için asc komutu kullanılmak
// zorunda da değildir.

 
select * from personel where cinsiyet='E' and ulke='Turkiye' order by maas asc
// Türkiyede yaşayan, cinsiyeti Erkek olan personelleri maaşlarına göre küçükten büyüğe
// sıralayıp getirir.
// 'asc' yazılmak zorunda değildir. Select zaten küçükten büyüğe sıralı getirecektir.
                                    -- TOP --

// Tanım: Getirilen verilerden en üstteki satırlardan kaç tane istenirse seçer.

select top 2 * from uyeler
// select ile tüm üyelerin seçildiği tablodan üstten ilk 2 satırı getirir.
                                  -- LIKE --

//Tanım: Aranacak metinsel içeriği, verilen pattern'e(arama deseni, kalıp) göre arar

select * from uyeler where soyisim like 'Z%'
// Soyismi Z harfi ile başkayan tüm üyeleri getirir.
// '%' ifadesi, tüm karakterler anlamına gelir.
// Yani bu sorguda, ilk harfi Z olan tüm üyeler

select * from uyeler where isim like '%er%'
// isminin içinde 'er' geçen tüm üyeleri getir.
// başta ve sonda '%' işareti olması, 'er' ifadesinin, ismin herhangi bir yerinde olabileceği
// anlamına gelir.

select * from uyeler where isim not like '%er%'
// içinde 'er' ifadesi geçMEyen kayıtları getirir. 'NOT', önüne getirilen şartın değilini alır.

select * from uyeler where isim like '%a'
// son harfi 'a' olan ancak 'a' harfinden önce bir tane karakter olanları getirir.
// Yani ismi 'a' ile başlayamaz.
// '_' karakteri bir tane karakteri ifade eder. Yani bu sorguda 'a' dan önce bir karakter zorunludur.

select * from uyeler where isim like 'M%m'
// ismi 'M' ile başlayıp 'm' ile biten kayıtları getirir.

select * from uyeler where email like 'h_zafer'
// Email bilgisinin ilk harfi 'h' olan, ikinci harfi herhangi bir karakter olabilen,
// ve 'zafer' ile biten kayıtları getirir

select * from uyeler where isim like '_a_'
// İsmi üç harfli olan, ikinci harfi 'a' olan, ilk ve son karakteri herhangi bir harf olanları
// getirir

select * from uyeler where soyisim like 'h_y_t'
// Soyisminin birinci harfi h, üçüncü harfi y, beşinci harfi t olan; 2. ve 4. harfleri herhangi
// bir harf olan üyeleri getirir.
// Örneğin 'hayat', 'huyut'

select * from uyeler where isim like 'er[hckmn]an'
// İsminin, üçüncü karakteri, [] içinde yazılan harflerden biri olan, son iki harfi 'an' olan 
// üyeleri listeler. Örneğin: Erhan, Erccan, Erkan, Erman

select * from uyeler where kullanici_adi like '[^CMK]in'
// [] içindeki karakterlerin başında '^' olduğunda, o karaktere sahip olmayan uyeleri getirir
// Örneğin; Cin, Min, Kin dışındaki bütün 'in' ile biten üç harfli kullanıcı adına sahip 
// üyeleri getirir.
                               -- IN -- 

// tanım: Verilen değerlerden kayıtla eşleşenleri getirir

select * from uyeler where yas in (22,23,24,25)
// Yaşı 22,23,24 veya 25 olan uyeleri getirir.

select * from uyeler where isim in('Ahmet', 'Mehmet', 'Ayşe')
// İsmi 'Ahmet' veya 'Mehmet' veya 'Ayşe' olanları getirir.
// Yani 'in' ifadesi, gerektiğinde 'or' yerine de kullanılabilir.
                      --  BETWEEN  --

// tanım: Verilen aralıktaki değerleri getirir

select isim, soyisim, yas from personel where yas between 20 and 30
// Yaşı 20 ve 30 aralığındaki personellerin isim, soyisim, yaş bilgilerini getirir
// Between ifadesinde verilen değerler de aralığa dahildir
                                 -- AS --

// tanım: Sorgulardaki tablo, sütun bilgilerine takma ad koyar.
// Sonuçlar getirildiğinde sütunlar artık o takma isim ile getirilir.
// Ancak bu, tablodaki orjinal sütun, tablo vs. adını etkilemez
// Sorgunun kalanında sütunlar, bu takma isim ile çağrılır. 
// Uzun sorgularda kullanmak mantıklıdır.

select isim as ad, soyisim as soyad from uyeler
// uyeler tablosundan isim alanını 'ad' olarak isimlendirerek, soyisim sütununu ise
// 'soyad' olarak isimlendirerek getirir. 

select u.isim, u.soyisim from uyeler u where u.cinsiyet='K' and u.isim='Derya'
// Cinsiyeti Kadın, isimi Derya olan uyelerin isim ve soyisim bilgilerini getirir.
// Sütun isimlerini yazarken başlarında tablonun(alias) isimlerini kullandık.
// Bunun nedeni, ileriki süreçte birden fazla tablo ile çalıştığımızda SQL'in,
// getirilecek verilerin hangi tablodan getirileceğinde kafasının karışmaması.
// Bunu yapmasak da çalışacaktı ama kullanmış olduk.
// Aliasing/takma ad yapmak için 'as' kullanmak zorunlu değildir.
// Takma ad vermek istedğimiz sütun ya da tablonun adından sonra direk takma adını
// yazdığımızda SQL bunu anlayacaktır.

use lab;
// lab veritabanına geçtik

create table personel(
	per_no int primary key,
	isim varchar(10),
	soyisim varchar(10),
	yas int,
	cinsiyet char(1),
	kent varchar(15),
	ulke varchar(15),
	maas int
);

insert into personel values(1, 'Ahmet', 'Yilmaz', 22, 'E', 'Ankara', 'Türkiye', 2000)

insert into personel values(2, 'Mehmet', 'Efe', 20, 'E', 'Ankara', 'Türkiye', 3000)

insert into personel values(3, 'Ayşe', 'Can', 23, 'K', 'İstanbul', 'Türkiye', 2500)

insert into personel values(4, 'Fatma', 'Ak', 35, 'K', 'Istanbul', 'Türkiye', 2800)

insert into personel values(5, 'Ahmet', 'Yalçın', 29, 'E', 'İstanbul', 'Türkiye', 2600)

insert into personel values(6, 'Ellen', 'Smith', 40, 'K', 'New York', 'ABD', 4000)

select * from personel where maas=1500 or maas=3000 or maas=2800 or maas=2600
// maaşı, yukarıda belirtilen personelleri getirir

select * from personel where maas in(1500, 3000, 2800, 2600)
// maaşı 1500,3000, 2800,2600 olanları getirir. Or ile tek tek yazmak yerine 'in'
// ifadesiyle istenilen değerleri yazabiliyoruz. Parametreler ile eşleşen satrıları
// getiriyor.

select * from personel where yas between 20 and 30 and cinsiyet = 'E'
// yaşı 20 ile 30 arasındaki erkek personelleri getirir

select * from personel where ulke = 'Turkiye' and isim like('A%') or isim like('F%')
and maas>1500 order by soyisim 
// Ülkesi Türkiye olan, ismi A ile başlayan ya da F ile biten ve maaşı 1500'den
// büyük olanları soyisimlerine göre alfabetik sıralayıp getirir

select * from personel where isim like('%s') or isim like('A%') and cinsiyet='K'
order by isim
// ismi A ile başlayıp ya da s ile biten erkekleri isme göre sıralayıp getirir.

select * from personel where isim like('A%s') order by isim
// İsmi A ile başlayıp s ile bitenleri isme göre sıralayıp getirir

select * from personel where isim like '_____' order by maas desc
// ismi beş harfli olanları maaslarına göre tersten sıralayıp getirir
// Like ifadesinin içine beş tane alt çizgi koyarak beş karakterden oluşmasını
// istediğimizi söyledik.

select * from personel where len(isim)=5
// ismi 5 karakterden oluşanları getirir
// Len fonksiyonu, içine yazılan stringin uzunluğunu verir.
// Yukarıdaki örnekte de ismi beş harften oluşanları 'like' ifadesi ile getirmiştik.
// Ancak ona gerek yoktur. Doğru olan 'len' fonksiyonunun kullanılmasıdır.

select isim, soyisim, yas from personel where ulke!= 'Turkiye' and maas>2000
and isim like('%[AEFaef]%') order by soyisim desc
// ulkesi Türkiye olmayan, maasşı 2000'dden büyük olan, isminde 'a,e,f' harflerinden
// biri bulunan kişilerin isim, soyisim, yas bilgilerini soyisimlerine göre ters
// sıralayarak getirir

select * from personel where ulke != 'Almanya' and (len(soyisim)>3 or soyisim like 'Y%')
and len(isim)>4 and yas<35
// Ülkesi Almanya olmayan, soyismi 3 karakterden uzun olan veya soyismi Y ile başlayan,
// ismi 4 harften uzun olan ve yası 35'ten küçük olanları getir

select isim, yas, maas*1.1 zamlimaas from personel where maas>yas*100 order by maas
// maaşı, yaşının 100 katından fazla olanların ismini, yaşını ve maaşının 10% zammlı
// halinin ismini zamlimaas yaparak ve maasa göre sıralayarak getirir.
// maaşı 1.1 ile çarparak zamlı halini getirmiş olduk. 
// maaşı 1.1 ile çarpıp yanina 'zamlimaas' yazınca maaşın zamnlı halini getirirlen
// sadece göstermelik olarak(orjinal sütun ismi değişmez) sütun ismini değiştirir
// 'maas*1.1 zamlimaas', 'maas*1.1 as zamlimaas' sorgusuna eşittir.
