create database VT_ALİSVERİS
use VT_ALİSVERİS

create table TblUrunKategori(
id int primary key identity(1,1),
kategori varchar(50) not null
);

create table TblUrunler (
id int primary key identity(1,1),
isim varchar(50) not null,
barkod char(13) not null,
fiyat float not null
urun_kategori_id int foreign key references TblUrunKategori(id)
);

create table TblMusteriAlisverisTarihi  {
id int primary key identity(1,1)
tarih datetime default current_timestamp
}
// Hafta 7'de olmayan yeni bir tablo oluşturduk. Bir müşteri bir alıiverişinde birden
// fazla ürün alabilir. Ancak TblAlisverisKayit tablosuna her satın alma eklendiğinde
// her biri farklı timestamp'a ve id'ye sahip oluyor. Yani sanki başka müşteriler
// alışveriş yapmış gibi. Halbuki sadece ortada bir müşterinin farklı ürün satın alma
// durumu var.

create table TblMusteriler
id int primary key identity(1,1),
isim varchar(30) not null,
soyisim varchar(30) not null,
ceptel char(10) not null,
mail varchar(30) not null,
);

create table TblAlisverisKayıt(
aliveris_id int foreign key references TblAlisverisKayitTarihi(id),
musteri_id int foreign key references Tblusteriler(id),
urun_id int foreign key references TblUrunler(id),
adet int not null
fiyat float not null
);
// Buradaki tarih sütunun sildik. Çünkü zaten MusteriAlisverisTarihi tablosundan tarih
// bilgisi gelecek.
// Geçen haftadan diğer bir farkı olarak; 'aliveris_id' sütunu 'primary key' iken şuanda
// 'foreign key' oldu. Çünkü 'TblAlisverisKayitTarihi' tablosunun id bilgisini burada
// 'aliveris_id' olark kullanıyoruz. Ki bu çok daha mantıklı. İşte bu yüzden yeni
// tablo oluşturduk.
// Önceki haftada müşterinin bir alışverişte her aldığı ürün için yeni bir 'aliveris_id'
// atanıyor. Ki bu oldukça yanlış.
// Bunun için yeni bir tabloda müşterinin alışveriş tarhini tutup her müşterinin
// alışveriş tarihine uniqe bir 'id' atayıp her alışverişi eşsiz ve eşsiz bir zaman
// yaptık. Müşteri artık birden çok ürün aldığında hepsinin 'aliveris_id' bilgisi aynı
// olacak çünkü hepsi aynı alışverişe ait.
// Bir de bu tabloya ekstra olarak 'fiyat' sütunu ekledik.
// Çünkü artık fiyat, alışverişe özgü olabilecek. Fiyat sonradan değişirse alışverişin
// içindeki satın alma fiyatını değiştirebileceğiz.
// !! Geçen hafta bu eklemeler yapıldığı için eski insert işlemlerine açıklamalar
// yazılmadı.

                  -- INNER JOIN ILE TABLOLARI BİRLEŞTİRME --

// tablo isimleri uzun olduğu için 'as' ile kısaltma kullanacağım

select avk.alisveris_id, mus.isim, mus.soyisim, urun.ad, urun.fiyat as güncel_fiyat,
avk.urun_fiyat alis_fiyati, avk.adet, avt.tarih
from TblAlisverisKayit as avk, TblMusteriler as mus, TblUrunler as urun,
TblMusteriAlisverisTarihi as avt, TblUrunKategori as ktg
where(
(mus.id=avk.musteri_id) and (urun.id=avk.urun_id)
and(avk.alisveris_id=avt.id) and(urun.kategori_id=ktg.id) 
)
// inner join ile yapabileceğimiz işi burada where şartı ile yaptık.
// Aşağıda ise inner join ile yapıyoruz

select avk.alisveris_id, mus.isim, mus.soyisim, urun.ad, urun.fiyat as güncel_fiyat,
avk.urun_fiyat alis_fiyati, avk.adet, avt.tarih
from (TblAlisverisKayit as avk
inner join TblMusteriler as mus on avk.musteri_id=mus.id
inner join TblUrunler as urun on avk.urun_id=urun.id
inner join TblMusteriAlisverisTarihi as avt on avk.alisveris_id=avt.id
inner join TblUrunKategori as ktg on urun.kategori_id=ktg.id
)
