// Veritabanında yapılabilen işlemler veritabanında yapılmalıdır.
// Örneğin 100.000 tane kullanıcının maaş bilgilerinin ortalamasını almak gerekiyorsa
// bunu veriitabanında yapmak mantıklıdır. Bu, elbette programlama dili ile verileri
// çektikten sonra yapılabilir. Ancak SQL'de daha hızlı olacaktır
// AVG: İçine yazılan sütunun ortalamasını alır.
// count: içine yazılan sütundaki null olmayan verilerin sayısını döner.

select avg(maas) ortalama_maas from personel
// personel tablosundaki kişilerin maaşlarının ortalamasını alıp, ortalama_maas takma adı
// ile getirir. Tablodaki tüm satırların ortalmasını alacaktır

select avg(maas) ortalama_maas from personel where isim like 'A%'
// ismi A ile başlayan personellerin maaşlarının ortalamasını alıp, ortalama_maas
// takma adı ile getirir.

select round(5.576, 2) as round_,
ceiling(5.579) as ceiling_,
floor(5.576) as floor_
// round fonksiyonu; içerisine girilen sayının virgülden sonraki parametre kadar
// basamak sayısı bırakır Yani sayımız 5.57 olacaktır
// ceiling fonksiyonu, içine girilen sayıyı bir üst tam sayıya yuvarlar.
// floor fonksiyonu; içine girilen sayıyı, bir alt tam sayıya yuvarlar.

select count(maas) as sayi from personel
// kaç tane maaş sütununa sahip satır olduğunu, 'sayi' takma adıyla getirir.
// eğer veriler arasında, değeri 'null' olan bir satır olsaydı null olan satır
// toplam sayıya dahil olmaz.
// select count(maas), isim from personel -> böyle bir sorgu hata verir. Çünkü
// count fonksiyonu döndürülmek istendiğinde count, maas verisine sahip bütün
// satırların toplam sayısını döndürmek isterken, isim ise isme sahip olan satırları
// döndürmek ister. Bu yüzden hata verir. 
// NOT: Getirilmek istenen verilerin her birinin getireceği satır sayısı aynı olmalıdır.

select count(*) as sayi from personel
// Her zaman maas sütunlarının hepsinin 'null' olmadığını bilemeyiz.
// Eğer maas verisine göre bir satır sayma işlemi yaparsak ve maas değeri null olan
// bir satır varsa bize yanlış satır sayısı dönecektir.
// Ancak '*' ifadesini koyarsak, sütunlar arasında
// illaki 'null' olmayan bir veri olacağı ve null olmayan verileri getireceği için
// bütün satırların sayısını dönecektir.

select count(maas) from personel where ulke='Turkiye'
// Ülkesi Türkiye olan personellerden maaş verisi 'null' olmayanları getirir.

select count(distinct ulke) as ulkeSayısı from personel
// disctinct; yanına yazılan sütunun içindeki verinin aynısından, tabloda
// birden çok sütunda varsa sadece bir tanesini döndürür. Bunu count fonksiyonu
// içine verdiğimizde toplam kaç ülke olduğunu döndürür.

select isim from personel where yas=(select max(yas) from personel)
// personel tablosundan yaşı en büyük olan personelin ismini getir.
// max() ve min() fonksiyonları, içine girilen sütundaki maximum/ minimum veriyi
// getirir.
// Where ifadesinden sonra bir alt sorgu daha yazılabilir. Çünkü zaten sorgular
// geriye bir değer döndürürler. Bu alt sorgudan dönen değeri alıp where şartımız
// için kullanabiliriz.
// Personellerin isimlerini çek. Nereden çek? Yaşı, personel tablosundaki yaşı en büyük
// olan personelin yaş bilgisine eşit olan kullanıcıyı çek.

select sum(maas) from personel
// personellerin maaş bilgislerini toplar ve getirir.
// Sum fonksiyonu, içine yazılan sütundaki verileri toplar.
                              -- GROUP BY --

// Tanım: Sütunu belirtilen aynı verileri gruplayarak getirir.

select ulke from personel group by ulke
// ulkeleri aynı olan personelleri ülkelerine göre gruplayıp toplam kaç tane farklı ülke 
// çıktıysa o ülkeleri getirir.
// Örneğin 150 kişinin ülkesi Türkiye, 500 kişinin ülkesi ABD, 120 kişinin ülkesi Almanya,
// 120 kişinin ülkesi İngiltere ise sadece 4 ülkenin ismini getirir. Çünkü 4 ülke altında
// gruplandı.
// Veriler gruplanırken hem en üstte hem en altta Trükiye varsa veriler
// getirilirken Turkiye grubu en altta olacaktır. Çünkü tablodaki son veri son grup olur

select ulke, avg(maas) from personel
// Bu sorgu hata verecektir. Getirilen verilerin satır sayıları aynı olmalıdır.
// Personel sayısı kadar ulke olmasına rağmen bir tane ortalama verisi olacaktır.

select ulke, avg(maas) from personel group by ulke
// Bu sorgu hata vermez. Çünkü gruplama yaptık. Cinsiyete göre verileri grupladık.
// Gruplanan veriler arasında ortalama alır. 4 tane grup varsa her grubun kendi içinde
// ortalamasını alır. Ve ulkelerini de getirir.

select ulke, kent from personel group by ulke, kent
// personelleri önce ülkelerine, sonra kentlerine göre gruplayıp ülke ve kent bilgilerini
// döndürür

select ulke, cinsiyet, avg(yas), count(*) from personel group by ulke, cinsiyet
// personelleri önce ülkeye sonra cinsiyete göre gruplayıp ülke, cinsiyet, yas
// ortalamalarını, ve her grupta kaç personel olduğunu döndürür. Eğer gruplama işlemi
// varsa, count grupların kendi içinde sayma yapar

select ulke, cinsiyet avg(yas), count(*) from personel where maas>2000
group by ulke, cinsiyet
// Maaşı 2000'den büyük olan personelleri ülke ve cinsiyetlerine göre sıralayıp ülke,
// cinsiyet, yas ortalamalarını,
// ve her grupta kaç personel olduğunu döndürür.
// Where ifadesi, group by'den önce gelmelidir.
                              -- HAVING --

// Tanım: Where ifadesine oldukça benzemektedir. Ancak where ifadesi sütunlar için
// çalışır Yani 'bana şu şartı sağlayan sütunları getir' diyebiliriz 'where' ile. 
// Ancak having'de birden fazla sütun birbiriyle ilişkiye giriyorsa, ortalamaları
// alınıyorsa, aralarında işlemler yapılıyorsa yani sadece bir sütun değil de
// birden fazla sütun arasındaki ilişkilerde şart arıyorsak o zaman having kullanılır.
// 'Group by' gibi ifadelerde kullanılabilir.
// Her sütunda tekrar eden bilgiler için having kullanılır.
// Mesela personellerin maaşları olsun. Biz ise ülkesi Türkiye olan personellerin
// maaş ortalamalarını almak istiyoruz. Her satırda 'ortalama' diye bir veri olmadığı
// için ortalama bilgisi, birden çok satırı ilgilendirdiği için where yerine
// having kullanılır burada.

select kent, ulke from personel group by kent having count(*) > 1
// personelleri kentlerine göre gruplayıp her kentin kendi içinde toplam sayısını
// alıp(yani o kente sahip kaç kişi olduğu)
// bu sayı 1'den fazla olan personellerin kent ve ulke bilgisini getirir.
// Burada where yerine having kullanıldı çünkü her kentten kaç kişi olduğu bilgisi
// her satırın sahip olduğu salt bir bilgi değil. Gruplandıktan sonra birden çok satırın
// sayılıp bulunduğu bir veri.

select avg(maas) from personel where cinsiyet='E' group by ulke having avg(maas)>3000
// personelleri ülkelerine göre gruplayıp cinsiyeti Erkek olan; her bir gruptaki, maaşı
// 3000'den büyük olan personellerin maaş ortalamalarını getirir.
// Burada cinsiyet için 'where' kullanıldı çünkü cinsiyet bilgisi her satırın sahip
// olduğu cinsiyet sütunundaki bir bilgidir. Her satır bu bilgiye sahiptir.
// Maaş ortalaması için ise 'having' kullanıldı çünkü maaş ortalama bilgisi satırların
// sahip olduğu bir bilgi değil. Gerekli satırlar arasında bir işlem yapılıp
// bulunan bir veridir.


select ulke, avg(yas) as yas_ort, avg(maas) as maas_ort, count(*) as kisi_sayisi
from personel where maas>=2000 group by ulke
having avg(yas)>=25 
// Maaşı 2000'e eşit veya daha fazla olan personelleri ülkelerine göre gruplayıp
// grubun yaş ortlaması 25'e eşit veya daha büyük olan grupların ülke, yaş ortalaması,
// maaş ortalaması ve kişi sayısı bilgilerini, gerekli olanlara 'as' ifadesi ile
// takma isim vererek getirir

select top(1) ulke, avg(maas) from personel group by ulke order by avg(maas) desc
// personelleri ulkelerine göre gruplayıp, grupları kendi içlerindeki maaş
// bilgilerine göre  her grubun ortalamasını alıp bunları ters sıralayıp gelen
// tablodaki en üst satırdaki verinin ulke ve maaş bilgisinin ortalamasını getirir.

select max(ort_maas) as max_ort_maas from (select ulke, avg(maas) as ort_maas
from personel group by ulke)
// personelleri ulkelerine göre gruplayıp, grupları kendi içlerindeki maaş bilgilerine
// göre sıralayıp grupların ülkelerini ve grupların ortalama maaşlarını, 'ort_maas'
// takma adıyla getirip gelen ortalama maaşların maximum'unu seçer ve getirir.
// Aslında bu, bir üstteki sorgu ile aynı sonucu döndürecektir. Ancak bu sorguda,
// ara sorgu(sorgu içinde sorgu) kullanılmıştır.
// Sırasıyla şöyle bir sıralama vardır:
// Ortalama maaşların(ort_maas) maximumunu , 'max_ort_maas' takma adıyla getir.
// Nereden getir?
// Personel tablosundaki personelleri ulkelerine göre sırala ve her grubun kendi
// içinde maaşlarını ortalamasını al. Grupların ülkelerinden ve 'ort_maas' takma adıyla
// isimlendirilmiş olan  maaş ortalamaları sonucu oluşan ara sorgudan dönen
// tablodan getir.
// (Son üç satırdaki cümle, ara sorgunun yaptığı işi ifade etmektedir.
// Ara sorgu bize aslında görünmeyen bir tablo döner. Biz normalde işlem yapmak
// istediğimiz tabloyu 'from' ifadesinden sonra yazarız. Burada ise ara sorgunun
// bize sağlamış olduğu görünmez tabloyu seçmiş oluyoruz. Bu örnekte bize,
// ulke ve ortalama maaş bilgilerinden oluşan bir tablo sağlanmış oldu ve bunun
// içinden seçim yapacağız. Zaten biz, normalde de yaptığımız sorgularda bize bir
// tablo döndüğünü biliyoruz. Örneğin SQL management Studio'da yaptığımız sorgular
// sonucunda ekranın altında bir tablo elde ediyoruz.)
// Ara sorguyu ileriki süreçte daha ayrıntılı göreceğiz.

select count(cinsiyet) from personel
// cinsiyet sütunu null(boş/verisi olmayan) olmayan satırların sayısını getirir

select count(distinct ulke) from personel
// Ulkesi aynı olmayan, ve ulke sütunu boş olmayan personel sayısını getirir
// Distinct, aynı olan verilerden sadece bir tanesini getirir. Yani ülkesi aynı
// olanların sadece bir tanesini döndürür. Kaç tane satır dönerse count da, bu
// dönen satırların kaç tane olduğunu söyler.

select ulke, kent from personel group by ulke, kent
// Önce ülkeye sonra kente göre gruplar. Gruplanan verileri getirir

select ulke, count(personel_no) personel_sayi from personel 
where ulke is not null group by ulke order by ulke, kent desc
// ülkesi null olmayan personelleri ulkelerine göre gruplayıp ulkeye göre sıralı,
// kente göre ters sıralı bir şekilde personel_no'su boş olmayan satırları
// personel_sayi takma adıyla ve ayrıca ulkelerini getirir.

update personel set cinsiyet='K' where personel_no=116
// personel_no'su 116 olan kullanıcının cinsiyetini Kadın olarak ayarla

select ulke, cinsiyet, count(*) from personel group by ulke, cinsiyet order by ulke, cinsiyet
// personelleri ulke ve cinsiyetlerine göre gruplayıp(ülkesi ve cinsiyeti aynı
// olanları gruplar) ardından ulke ve cinsiyete göre sıralayıp ulke, cinsiyet
// bilgilerini ve bunlardan kaç kayıt döndüğünü(count(*)) getirir.
// Eğer ulkeye göre gruplayıp cinsiyete göre gruplamazsa getirilecek satırlar
// arasında uyumsuzluk olacağından dolayı hata verecektir.
                               -- HAVING -- 

select ulke, cinsiyet, count(*) from personel
group by ulke, cinsiyet having count(*)>2
order by ulke, cinsiyet
// personelleri ulke ve cinsiyetlerine göre gruplar, ve her grubun içinde satır sayısı
// 2'den fazla olanları ülke ve cinsiyetlerine göre sıralayıp bunlardan ülke, cinsiyet
// ve her gruptaki satır sayısını getirir.

select ulke, avg(maas) from personel group by ulke
having min(maas)>=3000
// personelleri ülkelerine göre gruplayıp minimum maaşı 3000'den yüksek olan grupların
// ulke ve maaş ortalamalarını getirir.
                       --  ARA SORGU -- (nested query)

select ad, soyad, min(maas) from personel
// maaşı minimum olan kullanıcının ad ve soyadını getir.   X
// Yukarıdaki gibi bir sorgu olamaz. Hata verecektir
// Aslında biz burada demiş oluyoruz ki personellerin ad, soyadını ve maaşı minimum olan
// personeli getir.
// Ama maaşı minimum olan kullanıcı bir tane(ya da belki birden fazla) iken ad ve soyad
// ise bütün satırları dönecektir. Satır sayıları birbirinden farklı. Mantıksal
// olarak bir hata var

select ad, soyad from personel where maas<(select * min(maas) from personel)
// Maaşı, ortalama maaştan yüksek olan kullanıcıların ad ve soyadlarını getirir
// Ad ve soyadı getir. Nereden getir. Personel tablosundan getir. Peki kimi getireyim?
// Personel tablosundaki kullanıcılardan minimum maaş alan kullanıcıyı ara sorgu ile
// çek. Maaşı bu değerden küçük olan personelleri getir.
select upper(ad), soyad, maas*1.1 zamli_maas from personel
// ad bilgisini büyük harflerle, soyadı ve maaşın 1.1 katını zamli_maas ismiyle getirir
// Upper fonksiyonu, içine girilen değeri/ sütunu büyük harflerle yazar

select sum(maas) - (avg(maas)*count(*)) from personel
// personellerin maas ortalamasını, personel sayısı ile çarpıp personellerin maaş
// toplamından çıkarıp bize döndürür

select ulke from ulkeler where sehir in(select distinct kent from personel)
// ülkeler tablosundan ulkeyi getir. Nereden getir?
// şehri; personel tablosundaki personellerin kentlerinden birisi olanlardan getir.
// Daha ayrıntılı açıklarsak; 'distinct' ifadesi aynı verilerden sadece bir tanesini
// seçerdi. Personellerin şehri aynı olanlardan sadece bir satırı getirir.
// Böylece parantezin(ara sorgu) içinde elemanları eşsiz bir tablo döndürmüş
// oluyoruz aslında. 'In' ifadesi ile de şehri, bu listede olan ulkeleri getir
// demiş oluyoruz.

select * from personel where cinsiyet='E' and maas>(select avg(maas) from personel)
// cinsiyeti erkek olan; maaşı, ortalama maaştan yüksek olan personelleri getirir.
// Ara sorgu, personellerin ortalama maaşlarını döndürmemize yarar. Ve maaşı
// bu dönen veriden yüksek olanları getirecektir.
