create database VT_ALİSVERİS
use VT_ALİSVERİS

create table TblUrunKategori(
id int primary key identity(1,1),
kategori varchar(50) not null
);
// ürun kategorisini ürün tablosundan ayırmak önemlidir. Çünkü kategori ismi değişebilir
// ya da kategori silinebilir.
// Kategori ismi değiştirilirse, bu kategoriye sahip tüm ürünlerin kategori sütununda
// değişiklik yapmak zorunda kalacağız.
// Ya da ileride kategoriler görülmek istendiğinde
// ve kategoriler üzerinde işlemler yapılmak istenebilir.
// Ayrıca markette, temizlik kategorisine sahip tek bir ürün varsa ürünün silinmesi durumunda
// kategori de silinecektir.
// Buna maruz kalmamak için ayrıca bir table oluşturduk

create table TblUrunler (
id int primary key identity(1,1),
isim varchar(50) not null,
barkod char(13) not null,
fiyat int not null
urun_kategori_id int foreign key references TblUrunKategori(id)
);
// urun_kategori_id bir constraint'tir. TblUrunler tablosu ile TblUrunKategori
// tablosunu foreign key(tabancı anahtar) ile Urun kategorisinin 'id' bilgisine göre
// birleştirdik.
// foreign key de bir sütun olacak ve bu sütunun ismi 'urun_kategori_id', tipi int olur.
// Forign key oluşturuyoruz ama bu foreign key hangi tablonun hangi sütununu işaret edecek?
// Nereyi refere et? TblUrunKategori tablosundaki 'id' sütununu refere et.
// O tablodaki id sütununa göre ilişki kurar. Kategori tablosunun id'si bu tabloya sütun
// olarak eklenir ve böylece bu id yoluyla kategori tablosundaki, kategori tablosunun
// verilen id bilgisine sahip satıra ulaşılabilir.

create table TblMusteriler
id int primary key identity(1,1),
isim varchar(30) not null,
soyisim varchar(30) not null,
ceptel char(10) not null,
mail varchar(30) not null,
);

													-- INNER JOIN --

// Birden fazla tabloyu, önceden refere edilmiş değerlere göre birleştirir
// Yani; :)

select tblMusteriler.isim, tblMusteriler.soyisim, TblUrunler.ad, TblUrunler.fiyat,
TblAlisveris.tarih from
TblAlisverisKayit inner join TblMusteriler on TblAlisveris.musteri_id= TblMusteri.id),
inner join TblUrunler on TblAlisverisKayit.urun_id=TblUrunler.id
// tblMusteriler.isim, tbMusteriler.soyisim, TblUrunler.ad, TblUrunler.fiyat,
// TblAlisveris.tarih bilgilerini getir. Nereden getir?
// TblAlisveris tablosundaki musteri_id sütunundaki değer ile TblMusteriler tablosundaki
// 'id' değeri eşit olan müşterileri; TblAlisveris tablosundaki urun_id sütunundaki değer
// ile TblMusteriler tablosundaki 'id' değeri eşit olan ürünleri TblAlisverisKayıt
// tablosuna ekleyip o tablodan getir?

// Evet biraz karışık oldu. Şöyle sırasıyla gidelim!
// Inner Join'in nasıl kullanıldığını söylersek daha kolay olur:
// Hatırlayacağınız üzere bizim referans değerlerimiz vardı yani
// foreign key'lerimiz. Biz bunlar sayesinde tabloları birbirine bağlayabiliyorduk.
// Foreign key ile, eşleştirdiğimiz tablodaki 'id' değeri eşit olan satırı bize getirecekti
// Inner join söylediğimiz şeyi sağlıyor aslında.
// Referans ettiğimiz tablodaki id değeri ile foreign key eklediğimiz tablodaki foreign key
// değerini karşılaştırır. Eğer değerler eşleşiyorsa, refere ettiğimiz tablodaki
// eşleşen satırları getirip bizim ana tablomuzdaki satıra yeni sütunlar olarak ekler.
// Ancak bu, kalıcı bir ekleme değil. Sadece sonuçları getirdiğimiz andaki bir birleştirme
// Gerçek tablolara sütun eklemez.

// Örneğimiz üzerinden gidelim:
// Select ile belli alanların getirilmesini istedik. Ve getirilmesini istediğimiz
// sütunların başlarına tablo isimleri yazdık. Peki Neden? Inner join, işi gereği birden 
// çok tabloyu birbirine bağladığı için getirilecek sütunların hangi tabloda olduğunu
// bilemiyor(hele ki birden fazla tabloda aynı isme sahip sütunlar varsa. Mesela:
// isim sütunumuz birden fazla tabloda var. Eğer direk "'isim' sütununu getir" dersek SQL
// bize der ki? Tamam da hangi tablodaki 'isim' sütunun getireyim? İşte bunun önüne
// geçmek için tablo adı sütundan önce yazılır.)
// Önceden yaptığımız SQL sorgularında 'from' deyip ardından veri çekmek istediğimiz
// tabloyu yazardık. Ancak şimdi sadece bir tablo değil de 'inner join' sayesinde birden
// çok tablodan veri çekeceğimiz için yazımımız da biraz farklı oluyor haklı olarak.
// From dedikten sonra parantez içinde ana tablomuzun(TblAliverisKayit) ismini yazıyoruz.
// Ardından 'inner join' komutunu giriyoruz. Daha sonra bu tablo ile ilişkilendirmek
// (verilerini getirmek) istediğimiz tablonun ismini yazıyoruz. Bizim ilişkilendirmek 
// istediğimiz ilk tablomuz 'TblMusteriler' mesela.
// Bu 'inner join' işlemi neye göre yapılacak? Ana tablomuzun musteri_id sütunu ile
// TblMusteri(hedef tablo) tablosunun 'id' sütunu eşleşen satırları ana tablomuza ekleyecek
// 'on' komutu da buna yarıyor. 'Ben bu inner join işleimini yapıcam da neyin üzerinde yapayım?
// hangi sütuna göre yapayım?' Biz SQL'in bu sorusuna cevap vermiş oluyoruz. Bu işlemi,
// bu tablonun foreign key'i ile istediğim tablonun, istediğim sütunun eşleşenlere göre yap

// Bu bittikten sonra ikinci 'inner join'imizi yapıyoruz. Ana tablomuzu(TblAliverisKayıt)
// TblUrunler tablosu ile ilşkilendiriyoruz. Neye göre ilişkilendir? Ana tablomdaki urun_id
// sütunu ile TblUrunler tablosunun id sütunun eşleşenler getir.

// Yazımdaki parantezler kafanızı karıştırmasın. Parantezlerim mantığı ise şu: 
// From'dan sonra bir parantez açılır. Birinci inner join, hedef tablo ile birlikte parantez
// içine alınır. Diğer inner join'ler paranteze alınmayıp 'inner join' komutu ile başlar.
// Şu şekilde de yazımı mümkündür(yani parantezsiz olarak). Ancak hoca,
// Yukarıdakini gösterdiği için onu uygulamak daha doğru olur:

create table kitap(
kitap_id int primary key identity(100, 1),
kitap_ad varchar(30) not null,
yazar_id int,
yayinevi_id int,
sayfa_sayısı int,
fiyat int
)

create table yazar(
y_id int primary key identity(10, 10),
y_ad varchar(50),
y_yas int
)

create table yayinevi(
yevi_id int primary key identity(1, 1),
yevi_ad varchar(30),
yevi_adres varchar(20)
)

// Yukarıdak malum olduğu üzere üç tblo oluşturduk. Birbiriyle ilişkilendirmek için
// kitap tablona 'yayinevi_id', 'yazar_id' ekledik ve bu değerler foreign key olacak.
// Foregn key'leri ise birazdan constraint(kısıt) olarak ekleyeceğiz.

alter table kitap add constraint fk_yazar_kitap foreign key(yazar_id)
references yazar(y_id)

alter table kitap add constraint fk_yayinevi_kitap foreign key(yayinevi_id)
references yayinevi(yevi_id)
